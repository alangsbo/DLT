SELECT
utc.OWNER, utc.TABLE_NAME,
utc.COLUMN_NAME
, case WHEN utc.DATA_TYPE = 'DATE' THEN 'DATETIME2(7)' || CASE WHEN utc.NULLABLE = 'Y' THEN ' NULL' ELSE ' NOT NULL' END
 WHEN utc.DATA_TYPE ='NUMBER' AND utc.DATA_PRECISION IS NULL AND (utc.COLUMN_NAME LIKE '%AMOUNT%' OR utc.COLUMN_NAME LIKE '%BALANCE%' OR utc.COLUMN_NAME LIKE '%QUANTITY%' OR utc.COLUMN_NAME LIKE '%QTY%') AND utc.COLUMN_NAME NOT LIKE '%_ID' THEN 'NUMERIC(20,2)' || CASE WHEN utc.NULLABLE = 'Y' THEN ' NULL' ELSE ' NOT NULL' END
 WHEN utc.DATA_TYPE ='NUMBER' AND utc.DATA_PRECISION IS NULL AND utc.COLUMN_NAME LIKE '%RATE%' AND utc.COLUMN_NAME NOT LIKE '%_ID' THEN 'NUMERIC(20,2)' || CASE WHEN utc.NULLABLE = 'Y' THEN ' NULL' ELSE ' NOT NULL' END
 WHEN utc.DATA_TYPE ='NUMBER' AND utc.DATA_PRECISION IS NULL THEN 'NUMERIC(20,0)' || CASE WHEN utc.NULLABLE = 'Y' THEN ' NULL' ELSE ' NOT NULL' END 
 WHEN utc.DATA_TYPE ='NUMBER' AND utc.DATA_PRECISION IS NOT NULL THEN 'NUMERIC(' || cast(utc.DATA_PRECISION as varchar2(10)) || ',' || cast(utc.DATA_SCALE as varchar2(10)) || ')' || CASE WHEN utc.NULLABLE = 'Y' THEN ' NULL' ELSE ' NOT NULL' END 
 WHEN utc.DATA_TYPE ='CLOB' THEN 'VARCHAR(MAX)' || CASE WHEN utc.NULLABLE = 'Y' THEN ' NULL' ELSE ' NOT NULL' END 
 WHEN utc.DATA_TYPE ='LONG' THEN 'VARCHAR(MAX)' || CASE WHEN utc.NULLABLE = 'Y' THEN ' NULL' ELSE ' NOT NULL' END 
 WHEN utc.DATA_TYPE like '%TIMESTAMP%' THEN 'NVARCHAR(20)' || CASE WHEN utc.NULLABLE = 'Y' THEN ' NULL' ELSE ' NOT NULL' END
 WHEN utc.DATA_TYPE ='VARCHAR2' THEN 'VARCHAR('  || cast(utc.DATA_LENGTH as varchar2(10)) || ')' || CASE WHEN utc.NULLABLE = 'Y' THEN ' NULL' ELSE ' NOT NULL' END
 ELSE utc.DATA_TYPE || CASE WHEN utc.NULLABLE = 'Y' THEN ' NULL' ELSE ' NOT NULL' END 
END AS DataType
FROM
ALL_TAB_COLUMNS utc
WHERE
	utc.OWNER='%SCHEMANAME%'
	AND utc.TABLE_NAME='%TABLENAME%'
ORDER BY 
	utc.COLUMN_ID ASC
